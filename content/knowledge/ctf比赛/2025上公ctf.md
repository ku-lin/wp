```plain
# -*- coding: utf-8 -*-
import os
import re
from collections import Counter

# ====== 你只需要改这两个就够了 ======
EXPORT_DIR = os.path.join(os.getcwd(), r"C:\Users\zuziyi\Desktop\study\比赛\新生2\数据接口高频调用\90e3d693-a7d6-4e18-b72e-28601f48fd10\导出文件")  # 导出的文件夹：当前目录/导出文件
OUTPUT_UNIQUE_TXT = os.path.join(os.getcwd(), "unique_phones.txt")  # 去重手机号输出
# ====================================

# 11位手机号（以1开头），前后不能紧挨数字，避免把更长数字串截出来
PHONE_RE = re.compile(rb"(?<!\d)(1\d{10})(?!\d)")

# 可选：只扫描这些后缀（为空则扫描全部）
ONLY_SUFFIX = set()  # 例如：{".txt", ".json"}；留空=全部


def iter_files(root_dir: str):
    for base, _, files in os.walk(root_dir):
        for name in files:
            path = os.path.join(base, name)
            if ONLY_SUFFIX:
                ext = os.path.splitext(name)[1].lower()
                if ext not in ONLY_SUFFIX:
                    continue
            yield path


def main():
    if not os.path.isdir(EXPORT_DIR):
        print(f"[!] 找不到目录：{EXPORT_DIR}")
        print("    请确认：脚本所在目录下是否真的有一个叫“导出文件”的文件夹")
        return

    counter = Counter()
    file_count = 0
    bytes_total = 0

    for fp in iter_files(EXPORT_DIR):
        file_count += 1
        try:
            with open(fp, "rb") as f:
                data = f.read()
                bytes_total += len(data)
        except Exception as e:
            print(f"[!] 读取失败：{fp}  ({e})")
            continue

        # 提取手机号
        for m in PHONE_RE.findall(data):
            counter[m.decode("ascii", errors="ignore")] += 1

        # 可选：打印进度（文件很多时有用）
        if file_count % 200 == 0:
            print(f"[*] 已处理 {file_count} 个文件... 当前唯一手机号数：{len(counter)}")

    total_matches = sum(counter.values())
    unique_count = len(counter)

    print("\n====== 统计结果 ======")
    print(f"扫描目录：{EXPORT_DIR}")
    print(f"文件数量：{file_count}")
    print(f"数据总量：{bytes_total / (1024 * 1024):.2f} MB")
    print(f"手机号匹配总次数（含重复）：{total_matches}")
    print(f"手机号去重后数量：{unique_count}")

    # 输出去重列表（按数字排序）
    uniques_sorted = sorted(counter.keys())
    with open(OUTPUT_UNIQUE_TXT, "w", encoding="utf-8") as wf:
        for p in uniques_sorted:
            wf.write(p + "\n")

    print(f"\n[+] 去重手机号已写入：{OUTPUT_UNIQUE_TXT}")

    # 额外：如果你想看出现次数最多的前20个
    print("\n出现次数最多的前20个手机号：")
    for p, c in counter.most_common(20):
        print(f"{p}  x{c}")


if __name__ == "__main__":
    main()

```
