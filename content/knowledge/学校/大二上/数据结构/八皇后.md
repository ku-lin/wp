# 八皇后

皇后在行列、两个斜方向上不能共存

假设第一个皇后放在(1,1)

剩下的皇后的放置的位置可以依次找出来

只需要8次循环就行了。

#include<iostream>
#include<math.h>
#define SIZE 8
using namespace std;
int queens[SIZE]; // 皇后存放的位置，只需要存放每个位置的皇后存在第几行即可
void find_solutions(int row);
int sum=0; // 记录找到的解的个数

int main()
{
    find_solutions(0);
    cout<<sum<<endl;
    return 0;
}

{
    if (row == SIZE)
    { // 如果已经找到了8个皇后，则打印出来
    	sum++;

		for(int i=0;i<SIZE;i++){

        cout << endl;
        return;
    }
    for (int i = 0; i < SIZE; i++)
    {
        if (is_valid(row, i))
        { // 如果当前位置的皇后可以放置，则放置
            queens[row] = i;
            find_solutions(row + 1); // 递归调用，找下一个皇后位置
        }
    }
}

bool is_valid(int row, int col)
{
    // 判断当前位置的皇后是否可以放置
    for (int i = 0; i < row; i++)
    {
        if (queens[i] == col || abs(row - i) == abs(queens[i] - col))
        {
            return false;
        }
    }
    return true;
}
