```cpp
// RevPolishNotation.cpp : 此文件包含 "main" 函数。程序执行将在此处开始并结束。
//

#include <iostream>
#include <stdio.h>
#include <stdlib.h>


#define Init_STACK_CAP 2

struct stack
{
char* bottom = NULL;
char* top = NULL;
int size = 0;
};

int pushstack(stack* s, char d);

char popstack(stack* s);

char readstacktop(stack* s);

bool isstackEmpty(stack* s);

bool is_symbol(char d);

bool is_operator(char d);

int check_precedence(char a, char b);

int get_precedence_value(char d);

stack* create_stack();

void destroy_stack(stack* s);

int main()
{
    return 0;
}


stack* create_stack() {
    //创建栈空间
    stack* s = (stack * )malloc(sizeof(stack));
    //分配空间，失败则返回NULL
    s->bottom = (char*)malloc(Init_STACK_CAP * sizeof(char));
    if (NULL == s->bottom) {
        free(s);
        return NULL;
    }
    s->top = s->bottom;
    s->size = Init_STACK_CAP;

    return s;
}

void destroy_stack(stack* s) {
    free(s->bottom);
    free(s);
    return;
}

bool isstackEmpty(stack* s) {
    if (s->bottom == s->top) {
        return true;
    }
    return false;
}

char readstacktop(stack* s) {
    //检查栈状态    
    if (s->bottom == s->top) {
        //    printf("栈已空！无数据弹出！\n");
        return -1;
    }
    //读取栈顶元素
    return (s->top -1)[0];
}

bool is_symbol(char d) {
    if ((d >= 'a') && (d <= 'z')) {
        return true;
    }
    else if ((d >= 'A') && (d <= 'Z')) {
        return true;
    }
    else if ((d >= '0') && (d <= '9')) {
        return true;
    }
    return false;
}

bool is_operator(char d) {
    if (('+' == d) || ('-' == d) || ('*' == d) || ('/' == d) || ('^' == d)) {
        return true;
    }
    return false;
}


int pushstack(stack* s, char d) {
    //检查栈状态
    int curr_stack_size,i;
    char* old_buf = s->bottom;
    if ((s->top - s->bottom) >= s->size) {
        //     printf("栈满！等待栈空间扩大！\n");
        curr_stack_size = s->top - s->bottom;
        s->bottom = (char*)malloc(2 * s->size);
        if (NULL == s->bottom) {
            printf("栈扩容失败！\n");
            return -1;
        }
        for (i = 0; i < curr_stack_size; i++) {
            s->bottom[i] = old_buf[i];
        }
        free(old_buf);
        s->top = s->bottom + curr_stack_size;
        s->size = 2 * s->size;
        //     printf("当前栈大小：\t%d\n", s->size);
    }
    //压栈
    s->top[0] = d;
    s->top += 1;

    return 0;
}

char popstack(stack* s) {
    //检查栈状态    
    if (s->bottom == s->top) {
        //     printf("栈已空！无数据弹出！\n");
        return -1;
    }
    //弹栈
    s->top -= 1;
    return s->top[0];
}

//检查两个运算符的优先级，如果ab优先级一致，则返回0，如果a>b，则返回1，如果a<b则返回-1，如果其中有不是运算符的，返回-2
int check_precedence(char a, char b) {
    int precedenceA;
    int precedenceB;
    if (!is_operator(a) || !is_operator(b)) {
        printf("输入不全是运算符，无法比较运算符优先级！\n");
        return -2;
    }
    precedenceA = get_precedence_value(a);
    precedenceB = get_precedence_value(b);

    if (precedenceA == precedenceB) {
        return 0;
    }
    else if (precedenceA > precedenceB) {
        return 1;
    }
    return -1;
}

int get_precedence_value(char d) {
    //加减法类，优先级为1
    if (('+' == d) || ('-' == d)) {
        return 1;
    }
    //乘除法类，优先级为2
    if (('*' == d) || ('/' == d)) {
        return 2;
    }
    //指数运算，优先级为3
    if (('^' == d)) {
        return 3;
    }
    //未知类，优先级最低
    return 0;
}
```

### 逆波兰式：一种处理一个运算式的方法，又称为后缀表达式

比如说面对一个表达式：（a+b*c）/d^e*f

他的后缀表达是<font style="color:rgb(28, 31, 35);">a b c \* + d e ^ / f \*</font>

<font style="color:rgb(28, 31, 35);">面对所有运算符，将他们看为二元运算符</font>

*<font style="color:rgb(28, 31, 35);">ps：二元运算符指只需要两个元素的运算符</font>*

<font style="color:rgb(28, 31, 35);">所谓的前中后缀表达式就是将操作符放在两个元素的前、中、后面</font>

<font style="color:rgb(28, 31, 35);">中缀运算就是我们正常写法，将运算符放在两个元素的中心</font>

<font style="color:rgb(28, 31, 35);"></font>

<font style="color:rgb(28, 31, 35);">取出中间的b*c为例子，它的后缀就是bc*</font>

<font style="color:rgb(28, 31, 35);">之后将b\*c作为一个元素，直接用括号包起来（实际不需要，只是为了方便理解）</font>

<font style="color:rgb(28, 31, 35);">a+b*c就是a+（b*c）</font>

<font style="color:rgb(28, 31, 35);">转换为后缀表达式就是a（b\*c）+</font>

<font style="color:rgb(28, 31, 35);">之前已经转换过，所以可以直接把b*c代换出来写成abc*+</font>

<font style="color:rgb(28, 31, 35);"></font>

<font style="color:rgb(28, 31, 35);">继续处理，d^e转换就成了 de^</font>

<font style="color:rgb(28, 31, 35);">同样的方法，将（a+b*c）/d^e视为（a+b*c）/（d^e）</font>

<font style="color:rgb(28, 31, 35);">即（a+b\*c）（d^e）/</font>

<font style="color:rgb(28, 31, 35);">然后就可以直接将括号里面的转化出来，得到a b c \* + d e ^ /</font>

<font style="color:rgb(28, 31, 35);"></font>

<font style="color:rgb(28, 31, 35);">最后一步自己算，懒得写。。。。</font>

<font style="color:rgb(28, 31, 35);">注意其中的顺序，是根据我们的正常运算顺序运行的，即所有的内容都是根据运算符的先后级进行操作的。</font>

<font style="color:rgb(28, 31, 35);"></font>

<font style="color:rgb(28, 31, 35);">解释下来进行讨论如何使用栈对于数据进行维护</font>

1. 初始化一个空栈用于存储运算符，一个空列表用于存储后缀表达式
2. 遍历中缀表达式的每个元素
3. 根据元素类型执行不同操作：
   * 若为操作数，直接添加到后缀表达式列表
   * 若为左括号 "("，压入栈中
   * 若为右括号 ")"，弹出栈顶运算符并添加到后缀表达式，直到遇到左括号为止（左括号弹出但不加入结果）
   * 若为运算符，比较其与栈顶运算符的优先级：
     * 若栈顶运算符优先级更高或相等，弹出并添加到结果
     * 重复上一步直到栈顶运算符优先级更低或栈为空，再将当前运算符入栈
4. 遍历结束后，将栈中剩余运算符依次弹出并添加到结果

<font style="color:rgba(0, 0, 0, 0.85);">以中缀表达式 </font><code><font style="color:rgba(0, 0, 0, 0.85);">3 + 4 × 2 ÷ (1 - 5)</font></code><font style="color:rgba(0, 0, 0, 0.85);"> 为例：</font>

```plain
步骤 | 元素 | 栈操作           | 栈状态   | 后缀表达式
1   | 3    | 直接添加         | []      | [3]
2   | +    | 入栈             | [+]     | [3]
3   | 4    | 直接添加         | [+]     | [3,4]
4   | ×    | 优先级高于+，入栈 | [+,×]   | [3,4]
5   | 2    | 直接添加         | [+,×]   | [3,4,2]
6   | ÷    | 与×同级，弹出×    | [+]     | [3,4,2,×]
    |      | ÷入栈            | [+,÷]   | [3,4,2,×]
7   | (    | 入栈             | [+,÷,(] | [3,4,2,×]
8   | 1    | 直接添加         | [+,÷,(] | [3,4,2,×,1]
9   | -    | 入栈             | [+,÷,(,-] | [3,4,2,×,1]
10  | 5    | 直接添加         | [+,÷,(,-] | [3,4,2,×,1,5]
11  | )    | 弹出-            | [+,÷,(] | [3,4,2,×,1,5,-]
    |      | 弹出(不添加)     | [+,÷]   | [3,4,2,×,1,5,-]
12  | 结束 | 弹出÷            | [+]     | [3,4,2,×,1,5,-,÷]
    |      | 弹出+            | []      | [3,4,2,×,1,5,-,÷,+]
```

**注意括号其实并不算是运算符，括号因为优先级最高，可以无脑入栈**

**左括号可以视为栈底，即有右括号的时候一直弹栈到左括号**

总结一下

#### **<font style="color:rgb(79, 79, 79);">中缀表达式——>后缀表达式</font>**

<font style="color:rgb(77, 77, 77);">从前往后遍历：</font>

* <font style="color:rgb(51, 51, 51);">若为操作数，则直接输出</font>
* <font style="color:rgb(51, 51, 51);">若为(，则直接压栈</font>
* <font style="color:rgb(51, 51, 51);">若为)，则弹出栈顶元素，直到遇到(</font>
* <font style="color:rgb(51, 51, 51);">其他运算符，栈为空、栈顶为(、栈顶优先级低，则直接压栈；否则先弹出栈顶元素，再压入新元素</font>

#### **<font style="color:rgb(79, 79, 79);">中缀表达式——>前缀表达式（颠倒一下就行）</font>**

<font style="color:rgb(77, 77, 77);">从后往前遍历：</font>

* <font style="color:rgb(51, 51, 51);">若为数值，则直接输出</font>
* <font style="color:rgb(51, 51, 51);">若为)，则压入栈中</font>
* <font style="color:rgb(51, 51, 51);">若为(，则弹出栈顶元素，直到遇到)</font>
* <font style="color:rgb(51, 51, 51);">其他运算符，若栈为空、栈顶为)、比栈顶元素优先级高或相等，则直接压栈，否则弹出元素，再压入新元素</font>
* <font style="color:rgb(51, 51, 51);">反转最后结果</font>

```cpp
int main()
{
    char *exp = (char *)"a+b";
    stack *num = create_stack();
    stack *op = create_stack();
    int i = 0;
    while (exp[i] != NULL)
    {
        // 判断是不是数字，如果是就压入num栈
        if (is_symbol(exp[i]))
        {
            pushstack(num, exp[i]);
        }
        // 判断是不是运算符，如果是就压入op栈
        if (is_operator[exp[i]])
        {
            pushstack(num, popstack(op));
        }
        i++;
    }
    // op依次弹出到num栈
    while (!isstackEmpty(op))
    {
        char c = popstack(op);
        pushstack(num, c);
    }
    // 回显
    while (!isstackEmpty(num))
    {
        printf("%c", popstack(num));
    }
    destory_stack(num);
    destory_stack(op);
    return 0;
}
```

```cpp
int main()
{
    char *exp = (char *)"a+b";
    stack *num = create_stack();
    stack *op = create_stack();
    int i = 0;
    while (exp[i] != NULL)
        {
            // 判断是不是数字，如果是就压入num栈
            if (is_symbol(exp[i]))
            {
                pushstack(num, exp[i]);
            }
            // 判断是不是运算符，如果是就压入op栈
            if (is_operator[exp[i]])
            {
                //不断地判断当前栈顶的元素与i的优先级
                //当到了栈底或优先级较低时结束循环，将栈顶元素弹出到num栈，并将当前运算符压入op栈
                while (!isstackEmpty(op) && check_precedence(exp[i], readstacktop(op)) == 1)
                    {
                        char c = popstack(op);
                        pushstack(num, c);
                    }
                pushstack(op, exp[i]);
            }
            i++;
        }
    // op依次弹出到num栈
    while (!isstackEmpty(op))
        {
            char c = popstack(op);
            pushstack(num, c);
        }
    // 回显
    while (!isstackEmpty(num))
        {
            printf("%c", popstack(num));
        }
    destory_stack(num);
    destory_stack(op);
    return 0;
}
```

```cpp
int main()
{
    char *exp = (char *)"a+b*(c-d)";
    stack *num = create_stack();
    stack *op = create_stack();
    int i = 0;
    while (exp[i] != '\0')  // 修改为判断字符串结束符
        {
            // 判断是不是数字，如果是就压入num栈
            if (is_symbol(exp[i]))
            {
                pushstack(num, exp[i]);
            }
                // 判断是不是运算符，如果是就压入op栈
            else if (is_operator(exp[i]))  // 修改为else if
            {
                //不断地判断当前栈顶的元素与i的优先级
                //当到了栈底或优先级较低时结束循环，将栈顶元素弹出到num栈，并将当前运算符压入op栈
                while (!isstackEmpty(op) && check_precedence(exp[i], readstacktop(op)) == 1)
                    {
                        char c = popstack(op);
                        pushstack(num, c);
                    }
                pushstack(op, exp[i]);
            }
                // 处理左括号
            else if (exp[i] == '(')
            {
                pushstack(op, exp[i]);
            }
                // 处理右括号
            else if (exp[i] == ')')
            {
                while (!isstackEmpty(op) && readstacktop(op) != '(')
                    {
                        char c = popstack(op);
                        pushstack(num, c);
                    }
                if (!isstackEmpty(op) && readstacktop(op) == '(')
                {
                    popstack(op);  // 弹出左括号，但不压入num栈
                }
            }
            i++;
        }
    // op依次弹出到num栈
    while (!isstackEmpty(op))
        {
            char c = popstack(op);
            pushstack(num, c);
        }
    // 回显
    while (!isstackEmpty(num))
        {
            printf("%c", popstack(num));
        }
    destroy_stack(num);
    destroy_stack(op);
    return 0;
}
```
